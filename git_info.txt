3. git init - гит "разворачивается" в текущей папке
4. git rm --cached имя файла - удалить файл из ожидания
5. git log - просмотр логов
6. git log --oneline  - логи коротко в одну строку
7. git checkout id "лога" - просмотр версии данного лога
8. git checkout master - просмотр текущей версии
9. git revert id "лога" - отмена коммита
10. git reset id "лога" --hard - удаление коммитов которые были после данного коммита
11. :wq - выход из "странного" режима командной строки
