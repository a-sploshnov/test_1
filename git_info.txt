3. git init - гит "разворачивается" в текущей папке
4. git rm --cached имя файла - удалить файл из ожидания
5. git log - просмотр логов
6. git log --oneline  - логи коротко в одну строку
7. git checkout id "лога" - просмотр версии данного лога
8. git checkout master - просмотр текущей версии (переход на ветку)
9. git revert + id "лога" - отмена коммита
10. git reset + id "лога" --hard - удаление коммитов которые были после данного коммита
11. git branch + имя ветки - создание новой ветки
12. git branch -a - просмотор всех веток + узнать на какой сейчас ветке находишься
13. git merge + имя ветки - объединение веток: текущая + указанная в команде
14. git remote add origin + URL-адрес - подключение к репозиторию
15. git push -u origin + имя ветки - загрузка проекта из локальной версии в репозиторий
16. git clone + URL-адрес - скачивание проекта из репозитория в текущий каталог
17. git pull - скопировать из репозитория новые файлы (для работы более 1 чел)
:wq - выход из "странного" режима командной строки
